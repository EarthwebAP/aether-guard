Code Examples

Real-world usage patterns

LangChain

from aether_guard import guard
from langchain.chains import LLMChain

@guard(timeout=60, retries=3)
def reliable_chain():
    chain = LLMChain(llm=llm, prompt=prompt)
    return chain.run("Your prompt")

result = reliable_chain()

AutoGen

from aether_guard import guard
from autogen import AssistantAgent

@guard(timeout=120, retries=3)
def agent_workflow():
    agent = AssistantAgent("ai")
    return agent.initiate_chat(...)

result = agent_workflow()

Claude API

from aether_guard import guard
from anthropic import Anthropic

@guard(timeout=30, retries=3)
def call_claude():
    return client.messages.create(
        model="claude-3-sonnet",
        messages=[...],
    )

response = call_claude()

Async Functions

from aether_guard import async_guard

@async_guard(timeout=30, retries=3)
async def async_chain():
    response = await client.create(...)
    return response

result = asyncio.run(async_chain())
